    import type { ProFormInstance } from '@ant-design/pro-components';
    import { ProForm, ProFormDatePicker, ProFormText } from '@ant-design/pro-components';
    import { Button, message } from 'antd';
    import moment from 'dayjs';
    import { useRef } from 'react';
    

    const waitTime = (time: number = 100) => {
      return new Promise((resolve) => {
        setTimeout(() => {
          resolve(true);
        }, time);
      });
    };
    

    export default () => {
      const formRef = useRef<ProFormInstance>();
      const onFill = () => {
        formRef?.current?.setFieldsValue({
          name: '',
          company: '',
        });
      };
    

      const getCompanyName = () => {
        message.info(` "${formRef?.current?.getFieldValue('company')}"`);
      };
    

      const getFormatValues = () => {
        console.log('', formRef.current?.getFieldsFormatValue?.());
      };
    

      const validateAndGetFormatValue = () => {
        formRef.current?.validateFieldsReturnFormatValue?.().then((values) => {
          console.log('', values);
        });
      };
    

      return (
        <ProForm
          title=""
          formRef={formRef}
          submitter={{
            render: (props, doms) => {
              return [
                ...doms,
                <Button htmlType="button" onClick={onFill} key="edit">
                  
                </Button>,
                <Button htmlType="button" onClick={getCompanyName} key="read">
                  
                </Button>,
                <Button.Group key="refs" style={{ display: 'block' }}>
                  <Button htmlType="button" onClick={getFormatValues} key="format">
                    
                  </Button>
                  <Button htmlType="button" onClick={validateAndGetFormatValue} key="format2">
                    
                  </Button>
                </Button.Group>,
              ];
            },
          }}
          onFinish={async (values) => {
            await waitTime(2000);
            console.log(values);
            message.success('');
            return true;
          }}
        >
          <ProFormText
            width="md"
            name="name"
            label=""
            tooltip=" 24 "
            placeholder=""
          />
    

          <ProFormText width="md" name="company" label="" placeholder="" />
          <ProFormDatePicker name="date" initialValue={moment('2021-08-09')} />
        </ProForm>
      );
    };
