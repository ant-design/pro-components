      /** @name  */

      /**
       *  ReactNode  
       *
       * @name 
       */

        | ((
            schema: ProSchema<T, U, Extra>,
            type: ProSchemaComponentTypes,
            dom: React.ReactNode,
          ) => React.ReactNode)
        | React.ReactNode;
    

      /** @name  iconhover  */
      tooltip?: string | LabelTooltipType;

      tip?: string;
    

      render?: (
        dom: React.ReactNode,
        entity: T,
        index: number,
        action: ProCoreActionType,
        schema: ProSchema<T, U, Extra>,
      ) => React.ReactNode;
    

      /**
       * node value  onChange
       *
       * @name 
       */
      renderFormItem?: (
        item: ProSchema<T, U, Extra>,
        config: {
          index?: number;
          value?: any;
          onSelect?: (value: any) => void;
          type: ProSchemaComponentTypes;
          defaultRender: (newItem: ProSchema<T, U, Extra>) => JSX.Element | null;
        },
        form: FormInstance,
      ) => React.ReactNode;
